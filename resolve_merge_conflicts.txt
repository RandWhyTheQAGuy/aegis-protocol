# Resolve Merge Conflicts and Complete PR Merge

# 1. Clean up build artifacts to avoid checkout errors
rm -rf build/

# 2. Ensure you are in your project directory
cd /Users/rspickler/Documents/GitHub/aegis-protocol

# 3. Fetch the latest changes from origin
git fetch origin

# 4. Checkout the target branch (e.g., main or feature branch)
git checkout main
# or, if merging into a feature branch:
# git checkout feature/bft-clean-integrated-2ls

git pull origin main
# or, if on feature branch:
# git pull origin feature/bft-clean-integrated-2ls

# 5. Checkout your PR branch
git checkout fix/production-final

# 6. Rebase or merge the target branch into your PR branch
git rebase main
# or, if merging into a feature branch:
# git rebase feature/bft-clean-integrated-2ls

# 7. Resolve conflicts in the following files using your editor:
#   CMakeLists.txt
#   README.md
#   include/uml001/bft/remote_quorum_clock.h
#   include/uml001/core/policy.h
#   include/uml001/core/session.h
#   include/uml001/security/transparency_log.h
#   project_structure.txt
#   src/bft/remote_quorum_clock.cpp
#   src/core/passport.cpp
#   src/core/policy.cpp
# (Do NOT worry about build/ files, they are build artifacts and can be deleted.)

# 8. After resolving each conflict, stage the resolved files:
git add <resolved_file>

# 9. Continue the rebase (repeat steps 7-9 until all conflicts are resolved):
git rebase --continue

# 10. When the rebase is complete, push your branch (force push if rebased):
